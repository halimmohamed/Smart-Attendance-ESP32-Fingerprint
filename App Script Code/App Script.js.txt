function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet();
  var userSheet = sheet.getSheetByName(Users);
  var attendanceSheet = sheet.getSheetByName(Attendance);
  var action = e.parameter.action;
  
  if (action == login) {
    var id = e.parameter.id;
    var userData = userSheet.getDataRange().getValues();
    var userName = Unknown;
    var userGroup = NA;
    var userStatus = ;
    var userRow = -1;

   
    for (var i = 1; i  userData.length; i++) {

        userName = userData[i][1];
        userGroup = userData[i][2];
        userStatus = userData[i][3];
        userRow = i + 1;
        break;
      }
    }

    if (userStatus.toString().toLowerCase() == delete) {
      userSheet.deleteRow(userRow); 
      return ContentService.createTextOutput(COMMANDDELETE);
    }

    
    if (userName == Unknown  userName == ) {

      return ContentService.createTextOutput(New User Registered);
    }

    
    var today = new Date();
    today.setHours(0, 0, 0, 0);
    var attendanceData = attendanceSheet.getDataRange().getValues();
    var lastRowIndex = -1;

    for (var j = attendanceData.length - 1; j = 0; j--) {
      var rowDate = new Date(attendanceData[j][2]);
      rowDate.setHours(0, 0, 0, 0);
      if (attendanceData[j][1] == id && rowDate.getTime() == today.getTime()) {
        lastRowIndex = j + 1;
        break;
      }
    }

    var currentTime = new Date();
    
    if (lastRowIndex == -1) {
      
      attendanceSheet.appendRow([userName, id, new Date(), currentTime.toLocaleTimeString(), ---, userGroup]);
